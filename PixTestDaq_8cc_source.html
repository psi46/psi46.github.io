127 bool PixTestDaq::setTrgFrequency(uint8_t TrgTkDel){
128 
129  uint8_t trgtkdel = TrgTkDel;
130  double period_ns = 1 / (double)fParTriggerFrequency * 1000000; // trigger frequency in kHz.
131  double fClkDelays = period_ns / 25 - trgtkdel;
132  uint16_t ClkDelays = (uint16_t)fClkDelays; //debug -- aprox to def
133  uint16_t i = ClkDelays;
134 
135  //add right delay between triggers:
136  if (fParResetROC) {
137  fPg_setup.push_back(make_pair("resetroc", 15));
138  ClkDelays -= 15;
139  }
140  while (i>255){
141  fPg_setup.push_back(make_pair("delay", 255));
142  i = i - 255;
143  }
144  fPg_setup.push_back(make_pair("delay", i));
145 
146  //then send trigger and token:
147  fPg_setup.push_back(make_pair("trg", trgtkdel));
148  fPg_setup.push_back(make_pair("tok", 0));
149 
150  return true;
151 }

323 void PixTestDaq::doTest() {
324 
325  //Immediately stop if parameters not in range
326  if (fParOutOfRange) return;
327 
328  LOG(logINFO) << "PixTestDaq::doTest() start.";
329 
330  PixTest::update();
331  fDirectory->cd();
332 
333  fPg_setup.clear();
334 
335  //Set the ClockStretch

337 
338  //Set the histograms:
339  setHistos();
340 
341  //First send only a RES:
342  fPg_setup.push_back(make_pair("resetroc", 0)); // PG_RESR b001000
343  uint16_t period = 28;
344 
345  //Set the pattern generator:
346  fApi->setPatternGenerator(fPg_setup);
347 
348  fApi->daqStart();
349 
350  //Send only one trigger to reset:
351  fApi->daqTrigger(1, period);
352  LOG(logINFO) << "PixTestDaq::RES sent once ";
353 
354  fApi->daqStop();
355  //fApi->daqClear();
356 
357  fPg_setup.clear();
358  LOG(logINFO) << "PixTestDaq::PG_Setup clean";
359 
360  if(fParDelayTBM)
361  fApi->setTbmReg("delays",0x40); //FPix timing
362 
363  //Set the pattern wrt the trigger frequency:
364  LOG(logINFO) << "PG set to have trigger frequency = " << fParTriggerFrequency << " kHz";
365  if (!setTrgFrequency(20)){
366  FinalCleaning();
367  return;
368  }
369 
370  //Set pattern generator:
371  fApi->setPatternGenerator(fPg_setup);
372 
373  fDaq_loop = true;
374

376  fApi->daqStart();
377 
378  //If using number of triggers
379  if(fParNtrig > 0) {
380 
381  for (int i = 0; i < fParIter && fDaq_loop; i++) {
382  //Send the triggers:
383  fApi->daqTrigger(fParNtrig);
384  gSystem->ProcessEvents();
385  ProcessData(0);
386  }
387  fApi->daqStop();
388 
389  } else { //Use seconds
390 
391  //Start trigger loop + buffer fill management:
392  int finalPeriod = fApi->daqTriggerLoop(0); //Period is automatically set to the minimum by Api function
393  LOG(logINFO) << "PixTestDaq:: start TriggerLoop with period " << finalPeriod << " and duration " << fParSeconds << " seconds";
394 
395  //To control the buffer filling
396  uint8_t perFull;
397  uint64_t diff = 0, timepaused = 0, timeff = 0;
398  timer t;
399  bool TotalTime = false;
400 
401  while (fDaq_loop){ //Check every n seconds if buffer is full less then 80%
402  while (fApi->daqStatus(perFull) && perFull < 80 && fDaq_loop) { //Pause and drain the buffer if almost full.
403  timeff = t.get() - timepaused;
404  LOG(logINFO) << "Elapsed time: " << timeff / 1000 << " seconds.";
405  if (timeff / 1000 >= fParSeconds) {
406  fDaq_loop = false;
407  TotalTime = true;
408  break;
409  }
410  LOG(logINFO) << "buffer not full, at " << (int)perFull << "%";
411  gSystem->ProcessEvents();
412  ProcessData();
413  }
414  if (fDaq_loop){
415  LOG(logINFO) << "Buffer almost full, pausing triggers.";
416  fApi->daqTriggerLoopHalt();
417  diff = t.get();
418  ProcessData(0);
419  diff = t.get() - diff;
420  timepaused += diff;
421  LOG(logDEBUG) << "Readout time: " << timepaused / 1000 << " seconds.";
422  LOG(logINFO) << "Resuming triggers for " << fParSeconds - (timeff/1000) << " seconds.";
423  fApi->daqTriggerLoop(0);
424  }
425  else {
426  if (TotalTime) { LOG(logINFO) << "PixTestDaq:: total time reached - DAQ stopped."; }
427  fApi->daqStop();
428  ProcessData(0);
429  }
430  }
431  }
432 
433  PixTest::update();
434 
435  FinalCleaning();
436 
437  fApi->setClockStretch(0, 0, 0); //No Stretch after trigger, 0 delay
438 
439  LOG(logINFO) << "PixTestDaq::doTest() done";
440 }
