455 int PixTestPhOptimization::InsideRangePH(int po_opt, std::vector< std::pair<uint8_t, std::pair<uint8_t, std::vector<pxar::pixel> > > > &dacdac_max, std::vector< std::pair<uint8_t, std::pair<uint8_t, std::vector<pxar::pixel> > > > &dacdac_min){
456  //adjusting phscale so that the PH curve is fully inside the ADC range
457  int ps_opt = 999;
458  int maxPh(0);
459  int minPh(0);
460  bool lowEd=false, upEd=false;
461  int upEd_dist=255, lowEd_dist=255;
462  int safetyMargin = 50;
463  int dist = 255;
464  int bestDist = 255;
465  std::vector< std::pair<uint8_t, std::pair<uint8_t, std::vector<pxar::pixel> > > >::iterator dacit_max = dacdac_max.begin();
466  std::vector< std::pair<uint8_t, std::pair<uint8_t, std::vector<pxar::pixel> > > >::iterator dacit_min = dacdac_min.begin();
467  //or two for cycles??
468  LOG(logDEBUG) << "dacdac at max vcal has size "<<dacdac_max.size()<<endl;
469  LOG(logDEBUG) << "dacdac at min vcal has size "<<dacdac_min.size()<<endl;
470  while(dacit_max != dacdac_max.end() || dacit_min != dacdac_min.end()){
471  if(dacit_max->first == po_opt && dacit_min->first == po_opt && dacit_min->second.second.size() && dacit_max->second.second.size()) {
472  maxPh=dacit_max->second.second[0].value;
473  minPh=dacit_min->second.second[0].value;
474  lowEd = (minPh > safetyMargin);
475  upEd = (maxPh < 255 - safetyMargin);
476  upEd_dist = abs(maxPh - (255 - safetyMargin));
477  lowEd_dist = abs(minPh - safetyMargin);
478  dist = (upEd_dist > lowEd_dist ) ? (upEd_dist) : (lowEd_dist);
479  if(dist < bestDist && upEd && lowEd){
480  ps_opt = dacit_max->second.first;
481  bestDist=dist;
482  }
483  }
484  dacit_max++;
485  dacit_min++;
486  }
487  LOG(logDEBUG)<<"opt step 1: po fixed to"<<po_opt<<" and scale adjusted to "<<ps_opt<<", with distance "<<bestDist;
488  return ps_opt;
489 }
