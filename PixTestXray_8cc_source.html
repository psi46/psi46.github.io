98 bool PixTestXray::setTrgFrequency(uint8_t TrgTkDel) {
99  uint8_t trgtkdel = TrgTkDel;
100 
101  double period_ns = 1 / (double)fParTriggerFrequency * 1000000; // trigger frequency in kHz.
102  double fClkDelays = period_ns / 25 - trgtkdel;
103  uint16_t ClkDelays = (uint16_t)fClkDelays; //debug -- aprox to def
104 
105  // -- add right delay between triggers:
106  uint16_t i = ClkDelays;
107  while (i>255){
108  fPg_setup.push_back(make_pair("delay", 255));
109  i = i - 255;
110  }
111  fPg_setup.push_back(make_pair("delay", i));
112 
113  // -- then send trigger and token:
114  fPg_setup.push_back(make_pair("trg", trgtkdel)); // PG_TRG b000010
115  fPg_setup.push_back(make_pair("tok", 0)); // PG_TOK
116 
117  return true;
118 }
