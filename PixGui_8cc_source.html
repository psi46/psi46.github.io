266 void PixGui::handleButtons(Int_t id) {
267  if (id == -1) {
268  TGButton *btn = (TGButton *) gTQSender;
269  id = btn->WidgetId();
270  }
271 
272  switch (id) {
273  case B_DIRECTORY: {
274  LOG(logDEBUG) << Form("changing base directory: %s", fDirNameBuffer->GetString());
275  fOldDirectory = fConfigParameters->getDirectory();
276  if (0 == gSystem->OpenDirectory(fDirNameBuffer->GetString())) {
277  LOG(logINFO) << "directory " << fDirNameBuffer->GetString() << " does not exist, creating it";
278  int bla = gSystem->MakeDirectory(fDirNameBuffer->GetString());
279  if (bla < 0) {
280  LOG(logWARNING) << " failed to create directory " << fDirNameBuffer->GetString();
281  break;
282  }
283  }
284 
285  fConfigParameters->setDirectory(fDirNameBuffer->GetString());
286  changeRootFile();
287  break;
288  }
289  case B_FILENAME: {
290  LOG(logINFO) << Form("changing rootfilenme: %s", fRootFileNameBuffer->GetString());
291  changeRootFile();
292  break;
293  }
294  case B_EXIT: {
295  LOG(logDEBUG) << "PixGui::exit called";
296  CloseWindow();
297  break;
298  }
299 
300  case B_WRITEALLFILES: {
301  LOG(logDEBUG) << "PixGui::writeAllFiles called";
302  fConfigParameters->writeAllFiles();
303  break;
304  }
305  case B_POWER: {
306  if(fPower == true) {
307  fPower = false;
308  fbtnPower->ChangeBackground(fRed);
309  fbtnPower->SetText("Off");
310  fApi->Poff();
311  LOG(logDEBUG) << "Power set Off";
312  } else {
313  fPower = true;
314  fbtnPower->ChangeBackground(fGreen);
315  fbtnPower->SetText("On");
316  fApi->Pon();
317  LOG(logDEBUG) << "Power set On";
318  }
319  break;
320  }
321  case B_HV: {
322  if(fHV == true) {
323  hvOff();
324  } else {
325  hvOn();
326  }
327  break;
328  }
329  default:
330  break;
331  }
