776 bool PixTest::threshold(TH1 *h) {
777 
778  TF1 *f = fPIF->errScurve(h);
779 
780  double lo, hi;
781  f->GetRange(lo, hi);
782 
783  if (fPIF->doNotFit()) {
784  fThreshold = f->GetParameter(0);
785  // cout << " nofit fThreshold = " << fThreshold << endl;
786  fThresholdE = 1.;
787  fSigma = 0.5;
788  fSigmaE = 0.5;
789  } else {
790  h->Fit(f, "qr", "", lo, hi);
791  fThreshold = f->GetParameter(0);
792  // cout << " w/fit fThreshold = " << fThreshold << endl;
793  fThresholdE = f->GetParError(0);
794  fSigma = 1./(TMath::Sqrt(2.)*f->GetParameter(1));
795  fSigmaE = fSigma * f->GetParError(1) / f->GetParameter(1);
796  }
797 
798  fThresholdN = h->FindLastBinAbove(0.5*h->GetMaximum());
799 
800  if (fThreshold < h->GetBinLowEdge(1)) {
801  fThreshold = 0.;
802  fThresholdE = 0.;
803  fSigma = 0.;
804  fSigmaE = 0.;
805  fThresholdN = 0.;
806  return false;
807  }
808 
809  if (fThreshold > h->GetBinLowEdge(h->GetNbinsX())) {
810  fThreshold = h->GetBinLowEdge(h->GetNbinsX());
811  fThresholdE = 0.;
812  fSigma = 0.;
813  fSigmaE = 0.;
814  fThresholdN = fThreshold;
815  return false;
816  }
