63  fDoNotFit = false;

65  // -- determine step function and start of function range (to exclude spurious low-threshold readouts)
66  int STARTBIN(2);
67  int ibin(-1), jbin(-1);
68  double hmax(h->GetMaximum());
69  for (int i = STARTBIN; i <= h->GetNbinsX(); ++i) {
70  if (h->GetBinContent(i) > 0) {
71  ibin = i;
72  break;
73  }
74  }
75 
76  // require 2 consecutive bins on plateau
77  for (int i = STARTBIN; i < h->GetNbinsX(); ++i) {
78  if (h->GetBinContent(i) > 0.9*hmax && h->GetBinContent(i+1) > 0.9*hmax) {
79  jbin = i;
80  break;
81  }
82  }
83 
84  double lo = h->GetBinLowEdge(1);
85  // require 3 consecutive bins at zero
86  for (int i = 3; i < h->GetNbinsX(); ++i) {
87  if (h->GetBinContent(i-2) < 1 && h->GetBinContent(i-1) < 1 && h->GetBinContent(i) < 1) {
88  lo = h->GetBinLowEdge(i-2);
89  break;
90  }
91  }
92 
93  double hi = h->FindLastBinAbove(0.9*h->GetMaximum());
94 
95 
96  // -- setup function
97  TF1* f = (TF1*)gROOT->FindObject("PIF_err");
98  if (0 == f) {
99  f = new TF1("PIF_err", PIF_err, h->GetBinLowEdge(1), h->GetBinLowEdge(h->GetNbinsX()+1), 4);
100  f->SetParNames("step", "slope", "floor", "plateau");
101  } else {
102  f->ReleaseParameter(0);
103  f->ReleaseParameter(1);
104  f->ReleaseParameter(2);
105  f->ReleaseParameter(3);
106  f->SetRange(lo, hi);
107  }
108 
109  f->SetParameter(0, h->GetBinCenter((ibin+jbin)/2));
110  f->SetParameter(1, 0.2);
111  if (jbin == ibin) {
112  // cout << "XXXXXXXXXXX PixInitFunc: STEP FUNCTION " << h->GetTitle() << " ibin = " << ibin << " jbin = " << jbin << endl;
113  f->FixParameter(0, h->GetBinCenter(jbin));
114  f->FixParameter(1, 1.e2);
115  fDoNotFit = true;
116  }
117  f->FixParameter(2, 1.);
118  f->FixParameter(3, 0.5*h->GetMaximum());
119  return f;
120 }
