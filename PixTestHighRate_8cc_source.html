41 bool PixTestHighRate::setTrgFrequency(uint8_t TrgTkDel) {
42  uint8_t trgtkdel = TrgTkDel;
43 
44  double period_ns = 1 / (double)fParTriggerFrequency * 1000000; // trigger frequency in kHz.
45  double fClkDelays = period_ns / 25 - trgtkdel;
46  uint16_t ClkDelays = (uint16_t)fClkDelays; //debug -- aprox to def
47 
48  // -- add right delay between triggers:
49  uint16_t i = ClkDelays;
50  while (i>255){
51  fPg_setup.push_back(make_pair("delay", 255));
52  i = i - 255;
53  }
54  fPg_setup.push_back(make_pair("delay", i));
55 
56  // -- then send trigger and token:
57  fPg_setup.push_back(make_pair("trg", trgtkdel)); // PG_TRG b000010
58  fPg_setup.push_back(make_pair("tok", 0)); // PG_TOK
59 
60  return true;
61 }

193 
194  banner(Form("PixTestHighRate::xPixelAlive() ntrig = %d, vcal = %d", fParNtrig, fParVcal));
195  cacheDacs();
196 
197  fDirectory->cd();
198  PixTest::update();
199 
200  fApi->setDAC("ctrlreg", 4);

202 
203  fApi->_dut->testAllPixels(true);

205 
206  pair<vector<TH2D*>,vector<TH2D*> > tests = xEfficiencyMaps("highRate", fParNtrig, FLAG_CHECK_ORDER | FLAG_FORCE_UNMASKED);
207  vector<TH2D*> test2 = tests.first;
208  vector<TH2D*> test3 = tests.second;
209  vector<int> deadPixel(test2.size(), 0);
210  vector<int> probPixel(test2.size(), 0);
211  vector<int> xHits(test3.size(),0);
212  vector<int> vCalHits (test2.size(),0);
213  for (unsigned int i = 0; i < test2.size(); ++i) {
214  fHistOptions.insert(make_pair(test2[i], "colz"));
215  fHistOptions.insert(make_pair(test3[i], "colz"));
216  for (int ix = 0; ix < test2[i]->GetNbinsX(); ++ix) {
217  for (int iy = 0; iy < test2[i]->GetNbinsY(); ++iy) {
218  vCalHits[i] += static_cast<int>(test2[i]->GetBinContent(ix+1, iy+1));
219  // -- count dead pixels

221  ++probPixel[i];

223  ++deadPixel[i];
224  }
225  }
226  // -- Count X-ray hits detected
227  if (test3[i]->GetBinContent(ix+1,iy+1)>0){
228  xHits[i] += static_cast<int> (test3[i]->GetBinContent(ix+1,iy+1));
229  }
230  }
231  }
232  }
233 
234  copy(test2.begin(), test2.end(), back_inserter(fHistList));
235  copy(test3.begin(), test3.end(), back_inserter(fHistList));

237  h->Draw(getHistOption(h).c_str());
238  fDisplayedHist = find(fHistList.begin(), fHistList.end(), h);
239  PixTest::update();
240 
241  // -- summary printout
242  string deadPixelString, probPixelString, xHitsString, numTrigsString, vCalHitsString,xRayHitEfficiencyString,xRayRateString;
243  for (unsigned int i = 0; i < probPixel.size(); ++i) {
244  probPixelString += Form(" %4d", probPixel[i]);
245  deadPixelString += Form(" %4d", deadPixel[i]);
246  xHitsString += Form(" %4d", xHits[i]);
247  vCalHitsString += Form(" %4d",vCalHits[i]);
248  int numTrigs = fParNtrig * 4160;
249  numTrigsString += Form(" %4d", numTrigs );
250  xRayHitEfficiencyString += Form(" %.1f", (vCalHits[i]+fParNtrig*deadPixel[i])/static_cast<double>(numTrigs)*100);
251  xRayRateString += Form(" %.1f", xHits[i]/static_cast<double>(numTrigs)/25./0.64*1000.);
252  }
253 
254  LOG(logINFO) << "number of dead pixels (per ROC): " << deadPixelString;
255  LOG(logINFO) << "number of red-efficiency pixels: " << probPixelString;
256  LOG(logINFO) << "number of X-ray hits detected: " << xHitsString;
257  LOG(logINFO) << "number of triggers sent (total per ROC): " << numTrigsString;
258  LOG(logINFO) << "number of Vcal hits detected: " << vCalHitsString;
259  LOG(logINFO) << "Vcal hit detection efficiency (%): " << xRayHitEfficiencyString;
260  LOG(logINFO) << "X-ray hit rate [MHz/cm2]: " << xRayRateString;
261  LOG(logINFO) << "PixTestHighRate::doXPixelAlive() done";
262  restoreDacs();
263 }

360 void PixTestHighRate::maskHotPixels() {
361 
362  if (0 == fHitMap.size()) bookHist("maskHotPixels");
363 
364  int ntrig(1), vthrCompMin(5), vthrCompMax(105);
365  banner(Form("PixTestHighRate::maskHotPixels() running with ntrig = %d/step, vthrcomp = %d .. %d", ntrig, vthrCompMin, vthrCompMax));
366  cacheDacs();
367 
368  vector<uint8_t> vVthrComp = getDacs("vthrcomp");
369 
370  fDirectory->cd();
371 
372  fApi->_dut->testAllPixels(true);
373  fApi->_dut->maskAllPixels(false);
374 
375  vector<uint8_t> rocIds = fApi->_dut->getEnabledRocIDs();
376 
377  // -- Check and mask hot pixels
378  bool done(false);
379  for (int vThrComp = vthrCompMin; vThrComp <= vthrCompMax; vThrComp = vThrComp+10) {
380  LOG(logDEBUG) << "setting vthrcomp to " << vThrComp;
381  // set vthrcomp
382  done = true;
383  for (unsigned int iroc = 0; iroc < rocIds.size(); ++iroc){
384  done = done && (vThrComp > vVthrComp[iroc]);
385  fApi->setDAC("vthrcomp", vThrComp, rocIds[iroc]);
386  }
387  if (done) {
388  LOG(logDEBUG) << "all ROCs have original VthrComp < vthrcomp; breaking out of loop";
389  break;
390  }
391 
392  for (unsigned i = 0; i < fHitMap.size(); ++i) {
393  fHitMap[i]->Reset();
394  }
395 
396  doHitMap(1);
397  for (unsigned int i = 0; i < rocIds.size(); ++i) {
398  vector<pair<int,int> > hotPixels = checkHotPixels(fHitMap[i]);
399  for (unsigned int j = 0; j<hotPixels.size(); ++j) {
400  LOG(logDEBUG) << "mask hot pixel ROC/col/row: " << static_cast<int>(rocIds[i])
401  << "/" << hotPixels[j].first << "/" << hotPixels[j].second;
402  fApi->_dut->maskPixel(hotPixels[j].first, hotPixels[j].second, true, rocIds[i]);
403  fApi->_dut->testPixel(hotPixels[j].first, hotPixels[j].second, false, rocIds[i]);
404  }
405  }
406 
407  }
408 
409 
410  LOG(logINFO) << "# masked pixels per ROC:";
411  for (unsigned int i = 0; i < rocIds.size(); ++i) {
412  LOG(logINFO) << " ROC " << static_cast<int>(rocIds[i]) << ": " << static_cast<int>(fApi->_dut->getNMaskedPixels(rocIds[i]));
413  }
414 
415  restoreDacs();
416 }

421 void PixTestHighRate::doHitMap(int nseconds) {
422 
423  // -- setup DAQ for data taking
424  fPg_setup.clear();
425  fPg_setup.push_back(make_pair("resetroc", 0)); // PG_RESR b001000
426  uint16_t period = 28;
427  fApi->setPatternGenerator(fPg_setup);
428  fApi->daqStart();
429  fApi->daqTrigger(1, period);
430  fApi->daqStop();
431  fPg_setup.clear();
432  setTrgFrequency(50);
433  fApi->setPatternGenerator(fPg_setup);
434 
435  timer t;
436  uint8_t perFull;
437  fDaq_loop = true;
438 
439  fApi->daqStart();
440 
441  int finalPeriod = fApi->daqTriggerLoop(0); //period is automatically set to the minimum by Api function
442  LOG(logINFO) << "PixTestHighRate::doRateScan start TriggerLoop with period " << finalPeriod
443  << " and duration " << nseconds << " seconds";
444 
445  while (fApi->daqStatus(perFull) && fDaq_loop) {
446  gSystem->ProcessEvents();
447  if (perFull > 80) {
448  LOG(logINFO) << "Buffer almost full, pausing triggers.";
449  fApi->daqTriggerLoopHalt();
450  readData();
451  LOG(logINFO) << "Resuming triggers.";
452  fApi->daqTriggerLoop();
453  }
454 
455  if (static_cast<int>(t.get()/1000) >= nseconds) {
456  LOG(logINFO) << "Elapsed time: " << t.get()/1000 << " seconds.";
457  fDaq_loop = false;
458  break;
459  }
460  }
461 
462  fApi->daqTriggerLoopHalt();
463 
464  fApi->daqStop();
465  readData();
466  finalCleanup();
467
