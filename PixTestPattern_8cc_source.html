521 // ----------------------------------------------------------------------
522 void PixTestPattern::TriggerLoop(int checkfreq) {
523 
524  uint8_t perFull;
525  int nloop = 1;
526  uint64_t diff = 0, timepaused = 0, timeff = 0;
527  bool TotalTime = false;
528  timer t;
529  LOG(logINFO) << "PixTestPattern:: starting TriggerLoop for " << fParSeconds << " seconds";
530 
531  while (fDaq_loop)
532  {
533  if (nloop > 1){
534  diff = t.get() - diff;
535  timepaused += diff;
536  LOG(logDEBUG) << "PixTestPattern:: readout time " << timepaused / 1000 << " seconds";
537  LOG(logINFO) << "PixTestPattern:: restarting TriggerLoop for " << fParSeconds - (timeff / 1000) << " s";
538  }
539  //start triggerloop:
540  fPeriod = fApi->daqTriggerLoop(fParPeriod);
541  if (nloop == 1) LOG(logINFO) << "PixTestPattern:: TriggerLoop period = " << fPeriod << " clks";
542 
543  //check every checkfreq seconds if buffer is full less then 80%:
544  while (fApi->daqStatus(perFull) && perFull < 80 && fDaq_loop) {
545  mDelay(checkfreq * 1000);
546  timeff = t.get() - timepaused;
547  LOG(logINFO) << "PixTestPattern:: elapsed time " << timeff / 1000 << " seconds";
548  if (timeff / 1000 >= fParSeconds) {
549  fDaq_loop = false;
550  TotalTime = true;
551  break;
552  }
553  LOG(logINFO) << "PixTestPattern:: buffer not full, at " << (int)perFull << "%";
554  gSystem->ProcessEvents();
555  }
556 
557  if (fDaq_loop) {
558  LOG(logINFO) << "PixTestPattern:: buffer almost full, pausing triggers";
559  fApi->daqTriggerLoopHalt();
560  diff = t.get();
561  }
562  else {
563  if (TotalTime) { LOG(logINFO) << "PixTestPattern:: total time reached - DAQ stopped."; }
564  //fApi->daqTriggerLoopHalt();
565  fApi->daqStop();
566  }
567  // Get events and Print results on shell/file:
568  PrintEvents(fParSeconds, nloop, "loop");
569  nloop++;
570  }
571 }
